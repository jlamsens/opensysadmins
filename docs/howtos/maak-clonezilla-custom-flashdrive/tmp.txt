menuentry "NFS - RESTORE hpz2g1i LESPC - Dual boot" {
    linux /live/vmlinuz \
        boot=live union=overlay username=user hostname=lunar config quiet loglevel=0 noswap edd=on nomodeset enforcing=0 noeject locales="en_US.UTF-8" keyboard-layouts="NONE" \
        ocs_repository="nfs://172.16.21.251/home/guru/Downloads/DUMPS/hpz2g1i/" \
        ocs_live_run="ocs-sr --batch -g auto -e1 auto -e2 -q2 -r -j2 -k0 -scr -p poweroff restoredisk les-winlin-v2 nvme0n1 nvme1n1" \
        ip=172.16.11.252::172.16.11.254:255.255.255.0:lunar:eth0:static:8.8.8.8 net.ifnames=0 biosdevname=0 \
        vga=791 splash i915.blacklist=yes radeonhd.blacklist=yes nouveau.blacklist=yes vmwgfx.enable_fbdev=1

    initrd /live/initrd.img
}
