class ssh {
  # make sure the package is installed

}

        class ssh::config {
    # this file is for customizing ssh settings
    file {'/etc/ssh/sshd_config':
        ensure  => present,
        owner   => 'root',
        group   => 'root',
        mode    => '0600',
        notify  => Service['sshd'],
        source  => $osfamily ? {
        /(Ubuntu|Debian)/        => 'puppet:///modules/ssh/sshd_config',
        /(RedHat|CentOS|Fedora)/ => 'puppet:///modules/ssh/sshd_config_rpm',
        },
    }
    }
